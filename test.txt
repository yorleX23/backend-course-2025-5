# Запуск сервера у режимі розробки
npm run dev -- -h 127.0.0.1 -p 3000 -c ./cache

# PUT (завантаження локального файлу cat.jpg у кеш)
curl -X PUT --data-binary "@cat.jpg" http://127.0.0.1:3000/200

# GET (отримати з кешу або http.cat)
curl http://127.0.0.1:3000/200 --output 200.jpg

# DELETE (видалити з кешу)
curl -X DELETE http://127.0.0.1:3000/200

# GET після видалення (повторне завантаження з http.cat)
curl http://127.0.0.1:3000/200 --output 200_new.jpg
