#1. Запуск сервера 
#node index.js --host 127.0.0.1 --port 3000 --cache ./cache або npm run dev

# 2. GET (перше звернення)	
# Отримати картинку для коду 200 (сервер завантажить її з сайту http.cat)
curl http://127.0.0.1:3000/200 --output 200.jpg

#2. GET (повторне звернення)
# Тепер картинка має взятись із кешу, а не з http.cat
curl http://127.0.0.1:3000/200 --output 200_cached.jpg

# 3. PUT 
# Створити свою власну картинку 999.jpg у кеші
# (поклади поруч із index.js якийсь файл mycat.jpg, щоб команда його знайшла)
curl -X PUT --data-binary "@mycat.jpg" http://127.0.0.1:3000/999

#  4. GET після PUT 
# Перевір, що картинка 999 тепер зберігається в кеші
curl http://127.0.0.1:3000/999 --output newcat.jpg

#  5. DELETE 
# Видалити картинку 999 із кешу
curl -X DELETE http://127.0.0.1:3000/999

#  6. GET після DELETE 
# Спробуй отримати видалену картинку — має повернути Not Found
curl http://127.0.0.1:3000/999 --output deleted.jpg

# 7. GET для неіснуючого коду 
# Наприклад, код 999999 (його нема на http.cat) → має повернути Not Found
curl http://127.0.0.1:3000/999999 --output notfound.jpg
